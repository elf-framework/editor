function Main() {

    const [text, setText] = useState("title");

    // 렌더링 될 때마다 다시 실행
    useEffect(() => {
        console.log("rendering", text);
    }) // dependency 주입 하지 않음 

    // 최초 한번만 실행 
    useEffect(() => {
        console.log("rendering", text);
    }, []) // dependency 없음 

    // text 가 바뀔 때만 실행
    useEffect(() => {
        console.log("rendering", text);
    }, [text]) // text 주입     

    return <div>{text}</div>
}
