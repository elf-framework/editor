```typescript {6, 24-30} showLineNumbers
  export class Toolbar extends UIElement {
    props: ToolbarProps & DomEventType;
  }

  export interface ToolbarProps {
    items: ToolsProps[];
    align?: ToolbarAlignType;
    style: ToolbarStyle;
  }

  export type ToolbarAlignType =
    | "start"
    | "center"
    | "end"
    | "space-around"
    | "space-evenly";

  interface ToolsStyle {
    backgroundColor?: string;
    color?: string;
    height?: string;
  }    

  export interface ToolsProps {
    items: ToolsType[];
    style: ToolsStyle & CommonStyle;
    vertical?: boolean;
    emphasized?: boolean;
    moreIcon?: ContentType;
  }

  export type ToolsType =
    | ToolsItemType
    | ToolsMenuItemType
    | ToolsCustomItemType;

  export type ToolsItemType = {
    type: "item";
    title: string;
    icon?: string;
    selected?: boolean;
    selectedType?: "primary" | "secondary";
    events?: string[];
    onClick?: (event: PointerEvent) => void;
  };
  
  export type ToolsMenuItemType = {
    type: "menu";
    items?: ItemType[];
    title: ContentType;
    icon?: ContentType;
    selected?: boolean;
    events?: string[];
    opened?: boolean;
    trigger: "click" | "hover";
    direction?: MenuDirectionType;
    onOpen?: (event: PointerEvent, item: ToolsMenuItemType) => void;
    onClick?: (event: PointerEvent, item: ToolsMenuItemType) => void;
  };

```
